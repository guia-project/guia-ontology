#################################################################
# Restricciones estructurales mínimas
#################################################################

:Course rdfs:subClassOf [
    rdf:type owl:Restriction ;
    owl:onProperty :hasEvaluationSystem ;
    owl:minQualifiedCardinality "1"^^xsd:nonNegativeInteger ;
    owl:onClass :EvaluationSystem
] .

:Course rdfs:subClassOf [
    rdf:type owl:Restriction ;
    owl:onProperty :hasLearningOutcome ;
    owl:minQualifiedCardinality "1"^^xsd:nonNegativeInteger ;
    owl:onClass :LearningOutcome
] .

:EvaluationActivity rdfs:subClassOf [
    rdf:type owl:Restriction ;
    owl:onProperty :weight ;
    owl:minQualifiedCardinality "1"^^xsd:nonNegativeInteger ;
    owl:onDataRange xsd:decimal
] .

:ScheduleWeek rdfs:subClassOf [
    rdf:type owl:Restriction ;
    owl:onProperty :weekNumber ;
    owl:cardinality "1"^^xsd:nonNegativeInteger
] .
